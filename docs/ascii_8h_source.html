100  std::string Plot() {
101  // 1. calculate min and max
102  for (auto &label_trace_pair : series_) {
103  for (auto &item : label_trace_pair.second) {
104  min_ = std::min(item, min_);
105  max_ = std::max(item, max_);
106  }
107  }
108 
109  // 2. calaculate range
110  auto range = max_ - min_;
111 
112  // make basic padding as size of str(max)
113  basic_width_of_label_ = std::max(std::to_string((int)max_).length(),
114  std::to_string((int)min_).length());
115 
116  // 3. width and height
117  int width = 0;
118  for (auto &label_trace_pair : series_) {
119  width = std::max(width, (int)label_trace_pair.second.size());
120  }
121 
122  int legend_cols = 0, legend_rows = 0;
123  if (show_legend_) {
124  // determine width and height of legend, add to offset.
125  for (auto &label_trace_pair : series_) {
126  legend_rows++;
127  legend_cols =
128  std::max(legend_cols, (int)label_trace_pair.first.length());
129  }
130  }
131 
132  auto offset = offset_ + legend_cols;
133 
134  width += offset;
135 
136  if (std::isnan(height_)) {
137  height_ = range;
138  }
139 
140  // extend the height of the plot if we need more rows to display the
141  // legend than what the range of the data requires.
142  height_ = std::max((double)legend_rows, height_);
143 
144  // calculate ratio using height and range
145  auto ratio = height_ / range;
146 
147  int min2 = std::round(min_ * ratio);
148  int max2 = std::round(max_ * ratio);
149 
150  // 4. rows and cols of this chart
151  auto rows = max2 - min2;
152  auto cols = width;
153 
154  // 5. initialize chart using empty str
155  std::vector<std::vector<Text>> screen(
156  rows + 1, std::vector<Text>(cols, symbols_["empty"]));
157 
158  // 6. axis + labels
159  for (double y = min2; y <= max2; y++) {
160  auto label = FormatLabel(std::round(min_ + (y - min2) * range / rows));
161  // vertical reverse
162  screen[rows - (y - min2)][legend_cols] =
163  Text(label, Style().fg(Foreground::From(Color::BLUE)));
164  screen[rows - (y - min2)][offset - 1] =
165  Text((y == 0) ? symbols_["center"] : symbols_["axis"],
166  Style().fg(Foreground::From(Color::CYAN)));
167  }
168 
169  if (show_legend_) {
170  // 7. Legend
171  {
172  unsigned j = 0;
173  for (auto &label_trace_pair : series_) {
174  auto style = styles_[j % styles_.size()];
175  PutString(screen, label_trace_pair.first, style, j++, 0);
176  }
177  }
178  }
179 
180  // 8. Content
181  {
182  unsigned j = 0;
183  for (auto &label_trace_pair : series_) {
184  auto &trace = label_trace_pair.second;
185  auto style = styles_[j++ % styles_.size()];
186  auto y0 = std::round(trace[0] * ratio) - min2;
187  // vertical reverse
188  screen[rows - y0][offset - 1] = Text(symbols_["center"], style);
189 
190  for (size_t i = 0; i < trace.size() - 1; i++) {
191  auto y0 = std::round(trace[i] * ratio) - min2;
192  auto y1 = std::round(trace[i + 1] * ratio) - min2;
193 
194  if (y0 == y1) {
195  screen[rows - y0][i + offset] = Text(symbols_["parellel"], style);
196  } else {
197  screen[rows - y1][i + offset] =
198  Text(y0 > y1 ? symbols_["down"] : symbols_["up"], style);
199  screen[rows - y0][i + offset] =
200  Text(y0 > y1 ? symbols_["ldown"] : symbols_["lup"], style);
201  auto from = std::min(y0, y1);
202  auto to = std::max(y0, y1);
203  for (size_t y = from + 1; y < to; y++) {
204  screen[rows - y][i + offset] = Text(symbols_["vertical"], style);
205  }
206  }
207  }
208  }
209  }
210 
211  return Print(screen);
212  }
